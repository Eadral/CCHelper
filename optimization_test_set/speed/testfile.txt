const int MAX_NUM = 128 ;

int factorial(int n){
   if(n<=1) return (1);

   return(n*factorial(n-1)) ;
}

int mod(int x, int y){
   x=x-x/y*y;

   return (x) ;
}

void swap(int x, int y){
    int temp;

    printf("x = ", x) ;
    printf("y = ", y) ;
    temp = x;
    x=y;
    y=temp;
    printf("SWAP x = ", x) ;
    printf("SWAP y = ", y) ;
}

int full_num(int n, int j, int a){
  return (n*100+j*10+a) ;
}

int flower_num(int n, int j, int a){
  return (n*n*n+j*j*j+a*a*a) ;
}
